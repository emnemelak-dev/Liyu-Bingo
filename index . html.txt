  <div>የተሰበሰበ ገንዘብ፡ <span id="pool">0</span> ብር</div>

  <div id="numberDisplay" class="display-number bg-blue">ቁጥር፡ --</div>

  <button id="buyBtn" class="bg-green" onclick="buyCard()">ካርድ ግዛ (10 ብር)</button>

  <div id="board"></div>

  <button id="bingoBtn" class="bg-yellow" style="display:none; margin-top:10px;" onclick="claimBingo()">ቢንጎ! (Bingo)</button>

  <script>
    const socket = io();
    const tg = window.Telegram.WebApp;
    tg.expand();

    let myCard = [];

    // በአማርኛ ቁጥሮችን በድምፅ የመጥሪያ ተግባር (Speech Synthesis)
    function speakAmharic(text) {
      if ('speechSynthesis' in window) {
        const utterance = new SpeechSynthesisUtterance(text);
        utterance.lang = 'am-ET'; // የአማርኛ ድምፅ ቅንብር
        utterance.rate = 0.9; // ፍጥነት
        window.speechSynthesis.speak(utterance);
      }
    }

    socket.on('timerUpdate', (time) => {
      document.getElementById('timer').innerText = time;
    });

    socket.on('poolUpdated', (pool) => {
      document.getElementById('pool').innerText = pool;
    });

    function buyCard() {
      socket.emit('buyCard', tg.initDataUnsafe.user);
      document.getElementById('buyBtn').style.display = 'none';
    }

    socket.on('cardPurchased', (card) => {
      myCard = card;
      renderBoard(card);
      document.getElementById('bingoBtn').style.display = 'inline-block';
    });

    function renderBoard(card) {
      const boardDiv = document.getElementById('board');
      boardDiv.innerHTML = '';
      const colors = ['bg-green', 'bg-yellow', 'bg-red', 'bg-blue', 'bg-green'];

      for (let r = 0; r < 5; r++) {
        for (let c = 0; c < 5; c++) {
          const cell = document.createElement('div');
          cell.className = `cell ${colors[c]}`;
          cell.innerText = card[r][c];
          
          // በጣት ምልክት ማድረግ (Mark)
          cell.onclick = function() {
            this.classList.toggle('marked');
          };
          
          boardDiv.appendChild(cell);
        }
      }
    }

    // አዲስ ቁጥር ሲወጣ በአማርኛ መጥራት
    socket.on('newNumber', (num) => {
      document.getElementById('numberDisplay').innerText = "ቁጥር፡ " + num;
      speakAmharic("ቁጥር " + num);
    });

    socket.on('bingoWin', (data) => {
      alert(`🎉 ቢንጎ! ${data.type} ተሰርቷል። ያገኙት ሽልማት፡ ${data.prize} ብር`);
    });
  </script>
